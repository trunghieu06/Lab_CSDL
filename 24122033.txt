-- Chọn DB vừa tạo làm DB hiện hành
USE QLDT

CREATE TABLE KHOA (
	MAKHOA CHAR(5) CONSTRAINT PK_KHOA PRIMARY KEY,
	TENKHOA NVARCHAR(100),
	NAMTL INT,
	PHONG VARCHAR(10),
	DIENTHOAI VARCHAR(15),
	TRUONGKHOA CHAR(5),
	NGAYNHANCHUC DATE
);

CREATE TABLE BOMON (
	MABM CHAR(5) CONSTRAINT PK_BOMON PRIMARY KEY,
	TENBM NVARCHAR(100),
	PHONG VARCHAR(10),
	DIENTHOAI VARCHAR(15),
	TRUONGBM CHAR(5),
	NGAYNHANCHUC DATE,
	MAKHOA CHAR(5),
	CONSTRAINT FK_BOMON_KHOA FOREIGN KEY (MAKHOA) REFERENCES KHOA(MAKHOA)
);

CREATE TABLE GIAOVIEN (
	MAGV CHAR(5) CONSTRAINT PK_GIAOVIEN PRIMARY KEY,
	HOTEN NVARCHAR(100), 
	LUONG FLOAT,
	PHAI NVARCHAR(3),
	NGSINH DATE,
	DIACHI NVARCHAR(200),
	GVQLCM CHAR(5),
	MABM CHAR(5),
	CONSTRAINT FK_GIAOVIEN_BOMON FOREIGN KEY (MABM) REFERENCES BOMON(MABM)
);

ALTER TABLE KHOA
ADD CONSTRAINT FK_KHOA_GIAOVIEN FOREIGN KEY (TRUONGKHOA) REFERENCES GIAOVIEN(MAGV);

ALTER TABLE BOMON
ADD CONSTRAINT FK_BOMON_GIAOVIEN FOREIGN KEY (TRUONGBM) REFERENCES GIAOVIEN(MAGV);

ALTER TABLE GIAOVIEN
ADD CONSTRAINT FK_GIAOVIEN_GVQLCM FOREIGN KEY (GVQLCM) REFERENCES GIAOVIEN(MAGV);

CREATE TABLE GV_DT (
	MAGV CHAR(5),
	DIENTHOAI VARCHAR(15),
	CONSTRAINT PK_GV_DT PRIMARY KEY (MAGV, DIENTHOAI),
	CONSTRAINT FK_GV_DT_GIAOVIEN FOREIGN KEY (MAGV) REFERENCES GIAOVIEN(MAGV)
);

CREATE TABLE NGUOITHAN (
	MAGV CHAR(5),
	TEN NVARCHAR(100),
	NGSINH DATE,
	PHAI NVARCHAR(3),
	CONSTRAINT PK_NGUOITHAN PRIMARY KEY (MAGV, TEN),
	CONSTRAINT FK_NGUOITHAN_GIAOVIEN FOREIGN KEY (MAGV) REFERENCES GIAOVIEN(MAGV)
);

CREATE TABLE CHUDE (
	MACD CHAR(5) CONSTRAINT PK_CHUDE PRIMARY KEY,
	TENCD NVARCHAR(100)
);

CREATE TABLE DETAI (
	MADT CHAR(5) CONSTRAINT PK_DETAI PRIMARY KEY,
	TENDT NVARCHAR(100),
	CAPQL NVARCHAR(100),
	KINHPHI FLOAT,
	NGAYBD DATE,
	NGAYKT DATE,
	MACD CHAR(5),
	GVCNDT CHAR(5),
	CONSTRAINT FK_DETAI_CHUDE FOREIGN KEY (MACD) REFERENCES CHUDE(MACD),
	CONSTRAINT FK_DETAI_GIAOVIEN FOREIGN KEY (GVCNDT) REFERENCES GIAOVIEN(MAGV)
);

CREATE TABLE CONGVIEC (
	MADT CHAR(5),
	SOTT INT,
	TENCV NVARCHAR(100),
	NGAYBD DATE,
	NGAYKT DATE
	CONSTRAINT PK_CONGVIEC PRIMARY KEY (MADT, SOTT),
	CONSTRAINT FK_CONGVIEC_DETAI FOREIGN KEY (MADT) REFERENCES DETAI(MADT)
);

CREATE TABLE THAMGIADT (
	MAGV CHAR(5),
	MADT CHAR(5),
	SOTT INT,
	PHUCAP FLOAT,
	KETQUA NVARCHAR(100)
	CONSTRAINT PK_THAMGIADT PRIMARY KEY (MAGV, MADT, SOTT),
	CONSTRAINT FK_THAMGIADT_GIAOVIEN FOREIGN KEY (MAGV) REFERENCES GIAOVIEN(MAGV),
	CONSTRAINT FK_THAMGIADT_CONGVIEC FOREIGN KEY (MADT, SOTT) REFERENCES CONGVIEC(MADT, SOTT)
);
